static Uint8List open({
  required Key recipientKey,
  required Key senderKey,
  required Uint8List message,
  required Uint8List nonce,
}) {
  try {
    return askarKeyCryptoBoxOpen(
      recipientKey.handle,
      senderKey.handle,
      message,
      nonce,
    ).getValueOrException();
  } catch (e) {
    throw AskarKeyException('Error on crypto box open: $e');
  }
}
